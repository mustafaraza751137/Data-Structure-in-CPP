#include<bits/stdc++.h>
using namespace std;
void func(string s,string output)
{
	if(s.empty()){
		cout<<output<<endl;
		return;
	}
	for(int i=0;i<s.length();i++){
		func(s.substr(0,i)+s.substr(i+1),output+s[i]);
	}
}

int main() {
	string s;
	cin >> s;
    func(s,"");
}
